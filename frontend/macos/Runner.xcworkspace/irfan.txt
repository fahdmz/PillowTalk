insert into public.sleep_factors (
  user_id,
  name_key,
  updated_at
)
select distinct
  user_id,
  factor_key,
  now()
from seed_stats_rows
cross join lateral unnest(
  array_cat(
    array_remove(domains, 'sleep_substances'),
    sleep_substances
  )
) as factor_keys(factor_key)
on conflict (user_id, name_key) do update
  set updated_at = excluded.updated_at;

insert into public.sleep_factor_occurrences (
  factor_id,
  session_id,
  message_id,
  checkin_label_key,
  occurred_at,
  evidence_kind,
  source,
  confidence
)
select
  factor.id,
  row.session_id,
  row.user_message_id,
  case
    when row.checkin_mode = 'morning' then 'Morning Check-in'
    else 'Nightly Check-in'
  end,
  row.started_at + interval '1 minute',
  'user_reported',
  'local',
  row.confidence
from seed_stats_rows row
join public.sleep_factors factor
  on factor.user_id = row.user_id
 and factor.name_key = any(
   array_cat(
     array_remove(row.domains, 'sleep_substances'),
     row.sleep_substances
   )
 );

insert into public.session_recaps (
  session_id,
  title,
  summary,
  conclusion,
  dominant_emotion,
  domains,
  emotional_trend,
  sleep_observations,
  model_deployment,
  prompt_version,
  generated_at,
  updated_at
)
select
  session_id,
  '[seed] Catatan check-in ' ||
    case when checkin_mode = 'night' then 'malam' else 'pagi' end,
  '[seed] Ringkasan otomatis untuk pengujian statistik.',
  '[seed] Data pengujian - bukan rangkuman yang dihasilkan AI.',
  emotion,
  domains,
  jsonb_build_object(
    'direction',
    case
      when emotion in ('joy', 'love') then 'membaik'
      when emotion in ('neutral', 'surprise') then 'stabil'
      else 'memburuk'
    end,
    'observation',
    '[seed] Observasi otomatis untuk data pengujian.'
  ),
  coalesce((
    select jsonb_agg(observation)
    from (
      select 'Tidur sekitar ' || sleep_hours::text || ' jam semalam.' as observation
      where sleep_hours is not null
      union all
      select 'Bangun sekitar pukul ' || to_char(wake_time, 'HH24:MI') || '.'
      where wake_time is not null
      union all
      select 'Menyebutkan ' || array_to_string(sleep_substances, ', ') || ' di sekitar waktu tidur.'
      where cardinality(sleep_substances) > 0
    ) observations
  ), '[]'::jsonb),
  'seed-sql',
  'seed-v1',
  started_at + interval '6 minutes',
  started_at + interval '6 minutes'
from seed_stats_rows;

update public.sleep_factors factor
set
  level = case
    when counts.sample_size >= 3 then 'high'
    when counts.sample_size = 2 then 'medium'
    else 'low'
  end,
  updated_at = now()
from (
  select
    factor_id,
    count(*) as sample_size
  from public.sleep_factor_occurrences
  group by factor_id
) counts
where factor.id = counts.factor_id
  and factor.user_id = '5526818c-4a97-4fcc-a756-bfa70bf96316'
  and factor.name_key in (
    select distinct factor_key
    from seed_stats_rows
    cross join lateral unnest(
      array_cat(
        array_remove(domains, 'sleep_substances'),
        sleep_substances
      )
    ) as factor_keys(factor_key)
  );

commit;
